#include<iostream>
#include<math.h>
#define max 20
using namespace std;
class Prob{
    public:
    float var,mean,sum=0,dvn;
    int n;
    float arr[max];
    Prob(){
        cout<<"ENTER the size of data: ";
        cin>>n;
        cout<<"Enter "<<n<<"elemnets";
        for(int i=0;i<n;i++){
           cin>>arr[i];
        }
    }
    void calc(){
        for(int i=0;i<n;i++){
            sum+=arr[i];
        }
        mean = sum/n;
     for(int i=0;i<n;i++){
        var += pow(arr[i]-mean,2);
     }
     var/=n-1;
    dvn=sqrt(var);
    }
    void display(){
        cout<<"Mean"<<mean<<endl;
        cout<<"VAriance "<<var<<endl;
        cout<<"STandard"<<dvn;
    }
};
int main(){
    Prob P;
    P.calc();
    P.display();
}
